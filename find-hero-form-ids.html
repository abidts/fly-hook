// Find all form fields and their entry IDs
console.log('üîç Finding Google Form Field IDs...\n');

const fields = {};

// Find all input, textarea, and select elements with entry.* names
document.querySelectorAll('input[name^="entry."], textarea[name^="entry."], select[name^="entry."]').forEach(input => {
  // Try to find the label for this field
  let label = '';
  
  // Method 1: Check for associated label
  const labelElement = document.querySelector(`label[for="${input.id}"]`);
  if (labelElement) {
    label = labelElement.textContent.trim();
  }
  
  // Method 2: Check parent container for label
  if (!label) {
    const parent = input.closest('div[jsname]');
    if (parent) {
      const labelEl = parent.querySelector('.exportItemTitle, .labelText, .freebirdFormviewerWidgetLabel');
      if (labelEl) label = labelEl.textContent.trim();
    }
  }
  
  // Method 3: Check placeholder
  if (!label && input.placeholder) {
    label = input.placeholder;
  }
  
  // Method 4: Use field type as fallback
  if (!label) {
    label = `${input.type} field`;
  }
  
  fields[input.name] = label;
  console.log(`üìù ${label}: ${input.name}`);
});

// Also find data-item-id attributes
console.log('\nüìã Alternative method (data-item-id):');
document.querySelectorAll('[data-item-id]').forEach(item => {
  const labelEl = item.querySelector('.exportItemTitle, .labelText');
  if (labelEl) {
    const label = labelEl.textContent.trim();
    const itemId = item.getAttribute('data-item-id');
    if (label && itemId && !label.includes('Submit') && !label.includes('submit')) {
      console.log(`üìù ${label}: entry.${itemId}`);
    }
  }
});

// Output as JavaScript object for easy copying
console.log('\n‚úÖ Copy this to your HeroSlider.tsx:');
console.log('const GOOGLE_FORM_FIELDS = {');
Object.entries(fields).forEach(([key, label]) => {
  const cleanLabel = label.toLowerCase().replace(/[^a-z0-9]/g, '_');
  console.log(`  ${cleanLabel}: "${key}",  // ${label}`);
});
console.log('};');

// Show summary
console.log('\nüìä Summary:');
console.log(`Total fields found: ${Object.keys(fields).length}`);
console.log('Field IDs:', Object.keys(fields));
